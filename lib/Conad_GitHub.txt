# ================================
# ИНИЦИАЛИЗАЦИЯ НОВОГО ПРОЕКТА
# ================================

git init                 # создать локальный Git-репозиторий
git add .                # добавить все файлы в отслеживание
git commit -m "init"     # первый коммит (сохранение состояния)

# ================================
# ПРИВЯЗКА К GITHUB
# ================================
git remote add origin  https://github.com/kiplgcs/ESP-DASH_JeeUI/tree/cb657d9011f2ad32423c7ad6f0ee93c5684b52b9.git
git remote add origin  https://github.com/kiplgcs/ESP-DASH_JeeUI/work.git
git remote add origin https://github.com/USERNAME/REPO.git
                        # связать локальный проект с удалённым репозиторием GitHub

git remote -v           # проверить, к какому репозиторию подключён origin

# ================================
# ЕСЛИ НУЖНО УДАЛИТЬ НЕПРАВИЛЬНЫЙ ORIGIN
# ================================

git remote rm origin     # удалить старый/неверный origin (удалённый репозиторий)

# ================================
# ОТПРАВКА ПРОЕКТА НА GITHUB
# ================================

git push -u origin main  # отправить проект на GitHub впервые (создать main)
git push                 # отправлять изменения в дальнейшем

# ================================
# ОСНОВНЫЕ КОМАНДЫ В РАБОТЕ
# ================================

git status               # показать текущее состояние (что изменено/готово к коммиту)
git add .                # добавить все изменения
git commit -m "msg"      # сохранить изменения с комментарием
git push                 # отправить изменения в удалённый репозиторий
git pull                 # получить последние изменения с GitHub

# ================================
# ИСТОРИЯ И ПРОСМОТР
# ================================

git log                  # показать историю коммитов
git log --oneline        # краткая история
git diff                 # показать изменения перед коммитом

# ================================
# РАБОТА С ВЕТКАМИ
# ================================

git checkout -b new_branch  # создать новую ветку и перейти в неё

git checkout main        # переключиться обратно на основную ветку
git checkout work    - переключиться обратно на основную ветку

git branch               # показать список веток
git branch work - создать новую ветку
git push -u origin - work загрузить все файлы на GitHub В левом меню Git нажми на надпись "Publish Branch"

# ================================
# ПОЛЕЗНЫЕ КОМАНДЫ
# ================================

git clone URL            # клонировать чужой репозиторий
git restore FILE         # отменить изменения в файле
git reset HEAD FILE      # убрать файл из staged (после git add)
git fetch                # получить свежие данные без слияния
git merge BRANCH         # слить ветку с текущей
